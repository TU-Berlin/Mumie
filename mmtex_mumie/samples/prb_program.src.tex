\begin{prgsolution}
public static int getMin(int[] array) {
  int length = array.length;
  int min = Integer.MAX_VALUE;
  for ( int i = 0; i < length; ++i )
    min = Math.min(min, array[i]);          
  return min;
}
\end{prgsolution}
